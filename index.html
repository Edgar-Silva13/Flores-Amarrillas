import turtle as tur tur.speed(0) # Establece la velocidad máxima de dibujo def dibujar_circulo(color, radio): tur.fillcolor(color) tur.begin_fill() tur.circle(radio) tur.end_fill() tur.penup() tur.goto(0, -200) tur.pendown() # Dibuja el sol dibujar_circulo("yellow", 100) # Dibuja el césped tur.penup() tur.goto(-400, -200) tur.pendown() tur.fillcolor("green") tur.begin_fill() tur.forward(800) tur.right(90) tur.forward(400) tur.right(90) tur.forward(800) tur.right(90) tur.forward(400) tur.end_fill() # Dibuja la casa tur.penup() tur.goto(-100, -200) tur.pendown() tur.fillcolor("red") tur.begin_fill() for _ in range(4): tur.forward(200) tur.left(90) tur.end_fill() # Dibuja la puerta tur.penup() tur.goto(0, -200) tur.pendown() tur.fillcolor("brown") tur.begin_fill() tur.forward(50) tur.left(90) tur.forward(100) tur.left(90) tur.forward(50) tur.end_fill() # Dibuja la ventana tur.penup() tur.goto(75, -50) tur.pendown() tur.fillcolor("blue") tur.begin_fill() tur.forward(50) tur.left(90) tur.forward(50) tur.left(90) tur.forward(50) tur.left(90) tur.forward(50) tur.end_fill() tur.hideturtle() # Oculta la tortuga al finalizar tur.done()
